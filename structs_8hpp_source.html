93  bool is_type_matching(const std::string& path_to_key, const nl::json &value) const
94  {
95  // Types match directly
96  if (this->get_property_type(path_to_key) == value.type())
97  return true;
98  // Sometimes an unsigned/signed integer shall be assigned to an signed/unsigned value
99  // NOTE: This can happen if nlohmann::json interprets an signed value as being an unsigned value
100  if (this->get_property_type(path_to_key) == nl::json::value_t::number_integer && value.type() == nl::json::value_t::number_unsigned)
101  return true; // TODO: Check that unsigned value < signed positive max!
102  if (this->get_property_type(path_to_key) == nl::json::value_t::number_unsigned && value.type() == nl::json::value_t::number_integer)
103  return value >= 0U;
104  // The initial type is discarded, so we match anything
105  if (this->get_property_type(path_to_key) == nl::json::value_t::discarded)
106  return true;
107  // For dictionaries we have to handle the empty dict initialization case
108  if (this->get_property_type(path_to_key) == nl::json::value_t::object && value.type() == nl::json::value_t::null)
109  return true;
110  return false;
111  }
